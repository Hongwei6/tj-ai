## 常用运维命令

```bash
# SSH 连接
ssh T480s

# 查看所有容器状态
docker ps -a

# 查看天机学堂相关容器
docker ps -a --format 'table {{.Names}}\t{{.Status}}\t{{.Ports}}' | grep -E 'nacos|rabbitmq|elasticsearch|seata|xxl-job|mysql|redis'

# 查看容器日志
docker logs -f <容器名> --tail 100

# 重启服务
docker restart <容器名>

# 查看容器资源占用
docker stats --no-stream

# MySQL 备份单个数据库
docker exec 1Panel-mysql-qEJL mysqldump -uroot -phongwei <数据库名> > backup_<数据库名>.sql

# MySQL 恢复
docker exec -i 1Panel-mysql-qEJL mysql -uroot -phongwei <数据库名> < backup_<数据库名>.sql

# 批量重启所有天机学堂基础服务
docker restart nacos rabbitmq elasticsearch seata-server xxl-job-admin

# 清理未使用的 Docker 镜像（释放空间）
docker image prune -f

# 检查磁盘空间
df -h /

# 检查内存
free -h
```

## 构建与部署微服务

```bash
# 在项目根目录构建所有模块
mvn clean package -DskipTests

# 构建单个模块
mvn clean package -pl tj-gateway -am -DskipTests

# 上传 jar 包到服务器
scp tj-gateway/target/tj-gateway.jar T480s:/opt/tj-apps/

# 在服务器上启动服务
ssh T480s "java -jar /opt/tj-apps/tj-gateway.jar --spring.profiles.active=local"
```
